Работа с цветом

Уменьшение прозрачности

color: rgba(#......, 50%);

Преобразуется в rgba(..., ..., ..., 0.5)

Обесцвечивание

color: desaturate(#......, 20%);

Осветление

color: lighten(#......, 30%);

Затемнение

color: darken(#......, 10%);

Переменные

$name: ...;

.class_1 {
    property: $name;
}

.class_2 {
    property: $name + ... ;  (выражение)
}

Примеси

@mixin name() {
    property_1: value_1;
}

.class {
    @include name();
    property_2: value_2;
}

С переменными в качестве параметра

@mixin name_1($name_2) {
    property: $name_2;
}

name_3 {
    @include name_1(value);
}

Со значением по умолчанию

@mixin name_1($name_2: value_1) {
    property: $name_2;
}

name_3 {
    @include name_1(value_2);
}

name_4 {
    @include name_1();
}

у элемента name_4 свойство property будет иметь значение value_1 (значение по умолчанию), ибо параметру $name не передано значение аргумента 

@mixin button($width, $height, $text-size, $text-color, $bg-color) {
    width: $width;
    height: $height;
    font-size: $text-size;
    color: $text-color;
    background-color: $bg-color;
    text-align: center;
    line-height: $height;
    display: inline-block;
}

.button-menu {
    @include button(160px, 48px, 32px, black, red);
}

&

& указывает на текущий элемент 

.class {
    property_1: value_1;

    &:hover {
        property_2: value_2;
    }
}

Позволяет внутри набора правил описывать модификаторы
