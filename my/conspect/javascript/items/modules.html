Модули

Модуль-приём

Самовызывающаяся анонимная функция

(function name_1() {
    expression_1;
}());

Вызывается сразу после объявления.
Позволяет получить локальную область видимости.

Объектный интерфейс

Модуль записывается в переменную и в нём возвращаются методы доступные снаружи 

let name_1 = (function(){
  let name_2 = function() {    // снаружи не доступна
    expression_2;
  };
 
  return {
    name_3: name_2
  };
}());

name_1.name_3();   //-> выполнится expression_2

Modules ES6

export

export let name_1 = value_1;
let name_2 = value_2;

export {name_2};
let name_1 = value_1;
let name_2 = value_2;
  
export {name_1, name_2};
export function name_3() {
  expression_1;
}
function name_3() {
  expression_1;
}

export default name_3;      // Без {} при импортировании *

export default может быть только один в файле 

      // Экспорт с переименованием:

let name_1 = value_1;
let name_2 = value_2;
  
export {name_1 as n_1, name_2};     // name_1 переименовывается в n_1
      // export и export default **:

let name_1 = value_1;
let name_2 = value_2;

export {name_1};
export default name_2;

import

import name_3 from './name_js';     // Без {} ,т.к. export default *


    // то же самое, что и:
    
import {default as name_3} from './name_js';


    // можно сразу переименовать:
    
import n_3 from './name_js';        // name_3 переименуется в n_3
import {name_1, name_2, ... name_n} from './name_js';     // Импорт из файла name_js.js
      // ** :

import name_2, {name_1} from './name_js';

Импортирование с переименованием

import {name_1 as name_a} from './name_js';         // name_1 переименовывается в name_a
import * as name_a from './name_js';                // Все export из файла name_js.js переименовываются
                                                    // и становятся свойствами объекта name_a

Импортирование с деструктурированием

import {name_1, name_2, .... name_n} from './name_js';
