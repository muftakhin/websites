Свойства компонента

class Name_1 extends React.Component {
  render() {
    return (
      <element_1 {this.props.name_2} >
      {this.props.children}
      </element_1>
    );
  }
}

ReactDOM.render(
  <Name_1 name_2="value_2"/>,
  document.querySelector('css_selector_1')
);

в фигурных скобках { } содержимое оценивается как выражение
this.props.name_2 - доступ к заданному свойству с именем name_2 и значением value_2.
this.props - передаёт всё свойства дочернему элементу элемента Name_1.
this.props.children - вернёт дочерние элементы или просто содержимое будущего элемента, созданного методом .render() с использованием данного компонента. 

class Button extends React.Component {
  render() {
    return (
      <div>
        <button type={this.props.type}>
          {this.props.children}
        </button>
      </div>
    );
  }
}
            
ReactDOM.render(
  <Button type="submit">Отправить данные</Button>,
  document.querrySelector("css_selector_1")
);
