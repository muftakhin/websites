Свойства компонента

props

Используются для изменения генерируемых элементов.
Задаются при вызове компонента. 

    // Function:

function Name_1(props) {            // Можно деструктурировать объект props
  return (                          // и записать {name_1, name_2, ... name_n} вместо props
    <React.Fragment>
      .... {props.name_1} ....      // и писать просто name_1,
      .... {props.name_2} ....      // name_2,
      ....
      .... {props.name_n} ....      // name_n
    </React.Fragment>
  );
}

    // Class:

class Name_1 extends React.Component {
  render() {
    return (
      <element_1 {this.props.name_2} >
        {this.props.children}
      </element_1>
    );
  }
}

ReactDOM.render(<Name_1 name_1="value_1" name_2="value_2" .... name_n="value_n" />, document.querySelector('css_selector_1'));

В фигурных скобках { } содержимое оценивается как выражение.
this.props.name_2 - доступ к заданному свойству с именем name_2 и значением value_2.
this.props - передаёт всё свойства из Name_1 дочернему элементу element_1.
this.props.children - вернёт дочерние элементы или просто содержимое будущего элемента, созданного методом .render() с использованием данного компонента.
Имена свойств соответствующие именам стандартных атрибутов HTML-элементов рендерятся, а нестандартные нет.
Чтобы отрендерить свойство с нестандартным именем, нужно записать его data-"property_1", где property - не стандартное имя свойства.
Для передачи элементам динамических значений свойств (взятых из this.props ), фигурные скобки { } необходимо использовать внутри угловых скобок < >.
Динамические значения свойств определяются при создании компонента в его родителе. 

function Name_1(props) {
  return (
    <React.Fragment>
      .... {props.name_1} ....
      .... {props.name_2} ....
      ....
      .... {props.name_n} ....
    </React.Fragment>
  );
}

class Name_1 extends Component {
  constructor(props) {
    super(props);
  }

  render() {
    const {name_1, name_2, ... name_n} = this.props;
    return(
      <React.Fragment>
      .... {name_1} ....
      .... {name_2} ....
      ....
      .... {name_n} ....
    </React.Fragment>
    );
  }
}

const Name_2 = () => {
  return (
    <React.Fragment>
      <Name_1 name_1="value_11" name_2="value_12" .... name_n="value_1n" />
      <Name_1 name_1="value_21" name_2="value_22" .... name_n="value_2n" />
      <Name_1 name_1="value_31" name_2="value_32" .... name_n="value_3n" />
    </React.Fragment>
  );
};

ReactDOM.render(<Name_2 />, document.querySelector('css_selector_1'));
class HelloWorld extends React.Component {
  render() {
    return (
      <h1 {...this.props}>                       // Передача сразу всех свойств
        Hello {this.props.frameworkName} world!!!
      </h1>
    );
  }
}


ReactDOM.render(
  <div>
    <HelloWorld
      id='ember'
      frameworkName='Ember.js'
      title='A framework for creating ambitious web applications.'/>,
    <HelloWorld
      id='backbone'
      frameworkName= 'Backbone.js'
      title= 'Backbone.js gives structure to web applications...'/>
    <HelloWorld
      id= 'angular'
      frameworkName= 'Angular.js'
      title= 'Superheroic JavaScript MVW Framework'/>
  </div>,
  document.getElementById('content')
)
