Компонент React

Компонент это функция или класс, который записывается с большой буквы и возвращает код.
Компонент на основе функции не может хранить состояние компонента.
Компонент может возвращать JSX, null, boolean, массив объекта, Fragment, Портал и с 16-го React можно возвращать sting и number. 

    // Function:

const Name_1 = () => {
  return (
    jsx_code_1
  )
};


    // Class (при import React from 'react;)

class Name_1 extends React.Component {
  render() {
    return (
      jsx_code_1
    );
  }
}


// Class (при import React, {Component} from 'react';)

class Name_1 extends Component {
  render() {
    return (
      jsx_code_1
    );
  }
}



ReactDOM.render(<Name_1/>, document.querySelector('css_selector_1'));
