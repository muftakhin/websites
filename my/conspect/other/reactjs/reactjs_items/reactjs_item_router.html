Шаблон

import React from 'react';
import {BrowserRouter, Routes, Route, Link} from 'react-router-dom';

class Name_1 extends React.Component {
  ....
  render(){
    return(
      <BrowserRouter>
        <link to="/">Main</Link>
        <link to="/one">One Page</Link>
        <link to="/two">Two Page</Link>                     // У Link нет стандартного поведения как у <a>
        <Routes>
          <Route path="/" exact element={<NameMain />}/>    // exact - чтобы не случилась компоновка страницы
          <Route path="/one" element={<NameOne />}/>        // из совпадений с "/"
          <Route path="/two" element={<NameTwo />}/>
        </Routes>
      </BrowserRouter>
    );
  }
}

export default Name_1;
