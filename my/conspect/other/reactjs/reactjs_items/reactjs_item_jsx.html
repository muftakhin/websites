Имена компонентов начинаются с большой буквы, а теги элементов с маленькой

    // JSX:

class Name_1 extends React.Component {
  render() {
    return (
      <tag_name_1 property_11=value_11 property_12=value_12 >
        <Name_2 property_2=value_2 />
        <Name_3 property_3=value_3 />
      </tag_name_1>
    );
  }
}

    // JavaScript:
  
class Name_1 extends React.Component {
  render() {
    return React.createElement(
      "tag_name_1",
      {property_11: value_11, property_12: value_12},
      React.createElement(Name_2, {property_2: value_2}),
      React.createElement(Name_3, {property_3: value_3}),
    );
  }
}
    // JSX:

class HelloWorld extends React.Component {
  render() {
    return (
      <div>
        <h1>1. Hello world!</h1>
        <h1>2. Hello world!</h1>
      </div>
    );
  }
}

ReactDOM.render(
  <HelloWorld/>,
  document.getElementById('content')
)

    // JavaScript:

class HelloWorld extends React.Component {
  render() {
    return React.createElement(
      'div',
      null,
      React.createElement('h1', null, '1. Hello world!'),
      React.createElement('h1', null, '2. Hello world!')
    );
  }
}

ReactDOM.render(
  React.createElement(HelloWorld, null),
  document.getElementById('content')
)

Обработка выражений

class Name_1 extends React.Component {
  render() {
    return (
      <tag_name_1> content { Math.random()*100 } content </tag_name_1>
    );
  }
}

Все выражения, весь не статический контент необходимо заключать в фигурные скобки { }
Иначе он отобразится как текст.
В фигурные скобки { } заключаются переменные и переменные свойства

Возвращение нескольких элементов

Использование массивоподобного синтаксиса

class Name_1 extends React.Component { render() { return ( [ <tag_name_1 key="1">text_1</tag_name_1>, <tag_name_2 key="2">text_2</tag_name_2>, <tag_name_3 key="3">text_3</tag_name_3>, ] ); } }

key позволяет иметь дело с нужными элементами

Использование фрагментов React.Fragment

    // Полное указание:

class Name_1 extends React.Component {
  render() {
    return (
      <React.Fragment>
        <tag_name_1>text_1</tag_name_1>,
        <tag_name_2>text_2</tag_name_2>,
        <tag_name_2>text_3</tag_name_3>,
      </React.Fragment>
    );
  }
}
    // краткий синтаксис:

class Name_1 extends React.Component {
  render() {
    return (
      <>
        <tag_name_1>text_1</tag_name_1>,
        <tag_name_2>text_2</tag_name_2>,
        <tag_name_2>text_3</tag_name_3>,
      </>
    );
  }
}
