React JS

метод ReactDom.render()

ReactDom.render(jsx_code_1, place_1);

jsx_code_1 - элементы, которые нужно вывести
place_1 - место в DOM, где React должна отобразить эти элементы, например document.querySelector('css_selector_1')

Компонент React

class Name_1 extends React.Component {     // компонент
  render() {
    return jsx_code_1
  }
}

ReactDOM.render(
  <Name_1/>,
  place_1
);

Name_1 - имя компонента, с большой буквы

Свойства компонента

class Name_1 extends React.Component {
  render() {
    return (jsx_code_1 {this.props.name_2} jsx_code_2);
  }
}

ReactDOM.render(
  <Name_1 name_2="value_2"/>,
  place_1
);

в фигурных скобках { } содержимое оценивается как выражение
this.props.name_2 - доступ к заданному свойству с именем name_2 и значением value_2 

class Button_1 extends React.Component {
  render() {
    return (
      <div>
      <button type={this.props.button_2}>
        {this.props.children}
      </button>
    );
  }
}
            
ReactDOM.render(
  <Button_1 button_2="submit">Отправить данные</Button>,
  document.querrySelector("container")
);

this.prop.children - вернёт дочерние элементы или просто содержимое будущего элемента, созданного методом .render() с использованием данного компонента

Стилизация

При помощи CSS-классов

class Name_1 extends React.Component {
  render() {
    return (
      <tag_name_1 className="class_name_1" ...this.props.name_n... >
        {this.props.children}
      </tag_name_1>
    );
  }
}

ReactDOM.render(
  <Name_1 name_n="value_n">
    content_1
  </Name_1>,
  place_1
);

tag_name_1 - тег элемента
class_name_1 - имя класса
content_1 - содержимое элемента компонента, в т.ч. и дочерние элементы
